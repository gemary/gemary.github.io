Cú Pháp

khai báo biến : Kiểu dữ liệu rồi tên biến; int bien1; float bien2; char bien3;

In ra màn hình: printf("thứ gì đó"); (trong nháy kép sửa được)

Nhập 1 ký tự từ bàn phím: scanf("tùy kiểu dữ liệu",& tên của biến);

int (%d) // dùng chứa 1 giá số nguyên kiểu integer.

float (%f) // dùng chứa 1 giá trị kiểu số thực.

char (%c) // dùng để chứa 1 giá trị kiểu ký tự.

câu điều kiện: if( điều kiện ){ // code ở đây; } else{ // code ở đây;}

vòng lặp for: for(chạy từ; số lần lặp ; làm sai điều kiện ){ // code ở đây}
